52void print(Alignment position, std::string text, int blockWidth = {80})
53{
54 int spaces = 0;
55
56 switch (position)
57 {
58 case CENTRE:
59 spaces = ( blockWidth - static_cast<int>(text.size()) ) / 2;
60 break;
61
62 case RIGHT:
63 spaces = blockWidth - static_cast<int>(text.size());
64 break;
65 }
66
67 if (spaces > 0) std::cout << std::string (spaces, ' ');
68 std::cout << text << std::endl;
69}
